200    def set_lmin(self, lmin: Union[int, List[int]]):
201        """Take lmin parameter and set the corresponding attributes.
202
203        This handles automatically the case of a single value or a list of values. Note that the lmin for the cross-correlations is set to the geometrical mean of the lmin of the two fields when the likelihood approximation is not exact. This approximation has been tested and found to be accurate, at least assuming that the two masks of the two considered multipoles are very overlapped. On the other hand, lmin is set to the maximum of the two other probes for the exact likelihood. Indeed, the geometrical mean causes some issues in this case.
204
205        Parameters:
206            lmin (int or list):
207                Value or list of values of lmin.
208        """
209        self.lmins = {}
210        if isinstance(lmin, list):
211            assert (
212                len(lmin) == self.N
213            ), "If you provide multiple lmin, they must match the number of requested fields with the same order"
214            for i in range(self.N):
215                for j in range(i, self.N):
216                    key = self.fields[i] + self.fields[j]
217                    self.lmins[key] = int(max(lmin[i], lmin[j]))
218                    if self.like_approx != "exact":
219                        self.lmins[key] = int(np.ceil(np.sqrt(lmin[i] * lmin[j])))
220                    self.lmins[key[::-1]] = self.lmins[key]
221            self.lmin = min(lmin)
222        else:
223            self.lmin = lmin
224        return

277        """Produce fiducial spectra or read the input ones.

280        """

282        if self.cl_file is not None:

284                return self.cl_file
285            elif not self.cl_file.endswith(".pkl"):

287                    "The file provided is not a pickle file. You should provide a pickle file containing a dictionary with keys such as 'tt', 'ee', 'te', 'bb' and 'tb'."

292        try:
293            import camb
294        except ImportError:
295            print("CAMB seems to be not installed. Check the requirements.")
296
297        path = os.path.dirname(os.path.abspath(__file__))
298        planck_path = os.path.join(path, "planck_2018.ini")

303            pars.InitPower.set_params(
304                As=2.100549e-9,
305                ns=0.9660499,
306                r=self.r,

327    def get_noise_spectra(self):
328        """Produce noise power spectra or read the input ones.
329
330        If the user has not provided a noise file, this function will produce the noise power spectra for a given experiment with inverse noise weighting of white noise in each channel (TT, EE, BB). Note that you may want to have a look at the procedure since it is merely a place-holder. Indeed, you should provide a more realistic file from which to read the noise spectra, given that inverse noise weighting severely underestimates the amount of noise. If instead you provide the proper custom file, this method stores that.
331        """
332        if self.nl_file is not None:
333            if isinstance(self.nl_file, dict):
334                return self.nl_file
335            elif not self.nl_file.endswith(".pkl"):
336                print(
337                    "The file provided for the noise is not a pickle file. You should provide a pickle file containing a dictionary with keys such as 'tt', 'ee', 'te', 'bb' and 'tb'."
338                )
339                raise TypeError
340            with open(self.nl_file, "rb") as pickle_file:
341                return pickle.load(pickle_file)
342
343        print(
344            "***WARNING***: the inverse noise weighting performed here severely underestimates \
345            the actual noise level of LiteBIRD. You should provide an input \
346            noise power spectrum with a more realistic noise."
347        )
348
349        try:
350            import healpy as hp
351            import yaml
352            from yaml.loader import SafeLoader
353        except ImportError:
354            print("YAML or Healpy seems to be not installed. Check the requirements.")
355
356        assert (
357            self.experiment is not None
358        ), "You must specify the experiment you want to consider"
359        print(f"\nComputing noise for {self.experiment}")
360
361        path = os.path.dirname(os.path.abspath(__file__))
362        experiments_path = os.path.join(path, "experiments.yaml")
363        with open(experiments_path) as f:
364            data = yaml.load(f, Loader=SafeLoader)
365
366        instrument = data[self.experiment]
367
368        fwhms = np.array(instrument["fwhm"])
369
370        freqs = np.array(instrument["frequency"])
371
372        depth_p = np.array(instrument["depth_p"])
373        depth_i = np.array(instrument["depth_i"])
374
375        depth_p /= hp.nside2resol(self.nside, arcmin=True)
376        depth_i /= hp.nside2resol(self.nside, arcmin=True)
377        depth_p *= np.sqrt(hp.nside2pixarea(self.nside, degrees=False))
378        depth_i *= np.sqrt(hp.nside2pixarea(self.nside, degrees=False))
379
380        n_freq = len(freqs)
381
382        ell = np.arange(0, self.lmax + 1, 1)
383
384        keys = ["tt", "ee", "bb"]
385
386        sigma = np.radians(fwhms / 60.0) / np.sqrt(8.0 * np.log(2.0))
387        sigma2 = sigma**2
388
389        g = np.exp(ell * (ell + 1) * sigma2[:, np.newaxis])
390
391        pol_factor = np.array(
392            [np.zeros(sigma2.shape), 2 * sigma2, 2 * sigma2, sigma2],
393        )
394
395        pol_factor = np.exp(pol_factor)
396
397        G = []
398        for i, arr in enumerate(pol_factor):
399            G.append(g * arr[:, np.newaxis])
400        g = np.array(G)
401
402        res = {key: np.zeros((n_freq, self.lmax + 1)) for key in keys}
403
404        res["tt"] = 1 / (g[0, :, :] * depth_i[:, np.newaxis] ** 2)
405        res["ee"] = 1 / (g[3, :, :] * depth_p[:, np.newaxis] ** 2)
406        res["bb"] = 1 / (g[3, :, :] * depth_p[:, np.newaxis] ** 2)
407
408        res["tt"] = ell * (ell + 1) / (np.sum(res["tt"], axis=0)) / 2 / np.pi
409        res["ee"] = ell * (ell + 1) / (np.sum(res["ee"], axis=0)) / 2 / np.pi
410        res["bb"] = ell * (ell + 1) / (np.sum(res["bb"], axis=0)) / 2 / np.pi
411
412        res["tt"][:2] = [0, 0]
413        res["ee"][:2] = [0, 0]
414        res["bb"][:2] = [0, 0]
415
416        return res

526    def log_likelihood(self):
527        """Convert into log likelihood and sum over multipoles."""
528        if self.like_approx == "exact":
529            logp_ℓ = -0.5 * np.array(
530                get_chi_exact(
531                    N=self.N,
532                    data=self.data,
533                    coba=self.coba,
534                    lmin=self.lmin,
535                    lmax=self.lmax,
536                )
537            )
538        elif self.like_approx == "gaussian":
539            logp_ℓ = -0.5 * np.array(
540                get_chi_gaussian(
541                    N=self.N,
542                    data=self.data,
543                    coba=self.coba,
544                    mask=self.mask,
545                    inverse_covariance=self.inverse_covariance,
546                    lmin=self.lmin,
547                    lmax=self.lmax,
548                )
549            )
550        elif self.like_approx == "correlated_gaussian":
551            logp_ℓ = -0.5 * np.array(
552                get_chi_correlated_gaussian(
553                    # N=self.N,
554                    data=self.data,
555                    coba=self.coba,
556                    # mask=self.mask,
557                    inverse_covariance=self.inverse_covariance,
558                    # lmin=self.lmin,
559                    # lmax=self.lmax,
560                )
561            )
562        else:
563            print(
564                f"You requested some likelihood approximation (i.e. {self.like_approx}) which is not supported!"
565            )
566            raise KeyError
567

593            field = list(self.cobaCLS.keys())[1]

598            self.fields,
599            self.excluded_probes,
600            self.lmin,
601            self.lmax,
602            self.cobaCLS,
603            self.lmins,
604            self.lmaxs,
605        )
606
607        if self.debug:
608            ell = np.arange(0, self.lmax + 1, 1)
609            obs1 = 0
610            obs2 = 0
611            plt.plot(ell, self.fiduCOV[obs1, obs2, :], label="Fiducial CLs")
612            plt.plot(ell, self.cobaCOV[obs1, obs2, :], label="Cobaya CLs", ls="--")

615            plt.xlim(2, None)
616            plt.legend()
617            plt.show()

619        self.coba = (
620            self.cobaCOV[:, :, self.lmin : self.lmax + 1]
621            + self.noiseCOV[:, :, self.lmin : self.lmax + 1]
622        )
623
624        logp = self.log_likelihood()
625
626        if self.debug:
627            print(f"Log-posterior -->  {logp}")
628            exit()

329        """Produce noise power spectra or read the input ones.
330
331        If the user has not provided a noise file, this function will produce the noise power spectra for a given experiment with inverse noise weighting of white noise in each channel (TT, EE, BB). 334            if isinstance(self.nl_file, dict):
335                return self.nl_file

339                )

341            with open(self.nl_file, "rb") as pickle_file:
342                return pickle.load(pickle_file)

344        print(
345            "***WARNING***: the inverse noise weighting performed here severely underestimates \
346            the actual noise level of LiteBIRD. You should provide an input \
347            noise power spectrum with a more realistic noise."
348        )
349
350        try:
351            import healpy as hp
352            import yaml
353            from yaml.loader import SafeLoader
354        except ImportError:
355            print("YAML or Healpy seems to be not installed. Check the requirements.")
356
357        assert (
358            self.experiment is not None
359        ), "You must specify the experiment you want to consider"
360        print(f"\nComputing noise for {self.experiment}")
361
362        path = os.path.dirname(os.path.abspath(__file__))
363        experiments_path = os.path.join(path, "experiments.yaml")
364        with open(experiments_path) as f:
365            data = yaml.load(f, Loader=SafeLoader)
366
367        instrument = data[self.experiment]
368
369        fwhms = np.array(instrument["fwhm"])
370
371        freqs = np.array(instrument["frequency"])
372
373        depth_p = np.array(instrument["depth_p"])
374        depth_i = np.array(instrument["depth_i"])
375
376        depth_p /= hp.nside2resol(self.nside, arcmin=True)
377        depth_i /= hp.nside2resol(self.nside, arcmin=True)
378        depth_p *= np.sqrt(hp.nside2pixarea(self.nside, degrees=False))
379        depth_i *= np.sqrt(hp.nside2pixarea(self.nside, degrees=False))
380
381        n_freq = len(freqs)
382
383        ell = np.arange(0, self.lmax + 1, 1)
384
385        keys = ["tt", "ee", "bb"]
386
387        sigma = np.radians(fwhms / 60.0) / np.sqrt(8.0 * np.log(2.0))
388        sigma2 = sigma**2
389
390        g = np.exp(ell * (ell + 1) * sigma2[:, np.newaxis])
391
392        pol_factor = np.array(
393            [np.zeros(sigma2.shape), 2 * sigma2, 2 * sigma2, sigma2],
394        )
395
396        pol_factor = np.exp(pol_factor)
397
398        G = []
399        for i, arr in enumerate(pol_factor):
400            G.append(g * arr[:, np.newaxis])
401        g = np.array(G)
402
403        res = {key: np.zeros((n_freq, self.lmax + 1)) for key in keys}
404
405        res["tt"] = 1 / (g[0, :, :] * depth_i[:, np.newaxis] ** 2)
406        res["ee"] = 1 / (g[3, :, :] * depth_p[:, np.newaxis] ** 2)
407        res["bb"] = 1 / (g[3, :, :] * depth_p[:, np.newaxis] ** 2)
408
409        res["tt"] = ell * (ell + 1) / (np.sum(res["tt"], axis=0)) / 2 / np.pi
410        res["ee"] = ell * (ell + 1) / (np.sum(res["ee"], axis=0)) / 2 / np.pi
411        res["bb"] = ell * (ell + 1) / (np.sum(res["bb"], axis=0)) / 2 / np.pi
412
413        res["tt"][:2] = [0, 0]
414        res["ee"][:2] = [0, 0]
415        res["bb"][:2] = [0, 0]
416
417        return res
